üîßThe Tech Stack

  • Python 3.12 + NumPy + SciPy
  • FFmpeg for WAV‚ÜíMP3 encoding
  • Windows SAPI TTS for vocals
  • Every sound is math: sine waves for bass/pads, sawtooth via additive harmonics for leads, filtered white noise for drums/hats, convolution for reverb
  • ADSR envelopes shape every note (attack/decay/sustain/release)
  • Trance gates: amplitude modulation patterns (1/16th note chops)
  • Supersaw: 7 detuned sawtooth oscillators stacked for width
  • Granular synthesis: thousands of tiny sine grains scattered across stereo field
  • Vinyl crackle: filtered random impulses
  • Each track is a self-contained Python script, 200‚Äì300 lines

// supersaw_generator.py 
def supersaw(freq, duration, sr=44100, num_voices=7, detune=0.03):
    """7 detuned sawtooth oscillators stacked for massive width."""
    t = np.linspace(0, duration, int(sr * duration), endpoint=False)
    signal = np.zeros_like(t)
    for i in range(num_voices):
        # Spread voices evenly across detune range
        offset = (i / (num_voices - 1) - 0.5) * 2 * detune
        voice_freq = freq * (1 + offset)
        # Sawtooth via additive harmonics
        for h in range(1, 20):
            signal += (1/h) * np.sin(2 * np.pi * voice_freq * h * t)
    return signal / np.max(np.abs(signal))
// adsr_envelope.py 
def adsr(length, attack=0.01, decay=0.1, sustain=0.7, release=0.2, sr=44100):
    """Shape every note with attack/decay/sustain/release."""
    total = int(length * sr)
    a = int(attack * sr)
    d = int(decay * sr)
    r = int(release * sr)
    s = total - a - d - r
    envelope = np.concatenate([
        np.linspace(0, 1, a),         # Attack
        np.linspace(1, sustain, d),    # Decay
        np.full(s, sustain),           # Sustain
        np.linspace(sustain, 0, r),   # Release
    ])
    return envelope[:total]
